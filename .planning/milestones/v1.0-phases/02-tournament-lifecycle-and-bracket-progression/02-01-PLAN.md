export async function startTournament(req, res, next) {
  try {
    const { id } = req.params;
    const tournament = await tournamentLifecycleService.startTournament(id);
    return res.status(200).json({ success: true, data: { id: tournament.id, status: tournament.status } });
  } catch (err) {
    if (err.statusCode) {
      return res.status(err.statusCode).json({
        success: false,
        error: { code: err.code, message: err.message }
      });
    }
    next(err);
  }
}

  <action>
Create a new ES module service file. Export three functions:

**startTournament(tournamentId)**
- Reads tournament by id (select: id, status only)
- Throws 404 TOURNAMENT_NOT_FOUND if not found (use the same throwError helper pattern as matchResultService.js)
- Throws 400 INVALID_STATUS_TRANSITION if status !== 'SCHEDULED' (message: `Cannot start a tournament with status ${tournament.status}`)
- Updates tournament: status = 'IN_PROGRESS', lastStatusChange = new Date()
- Returns the updated tournament record
- Note: registration closure is automatic — tournamentRegistrationService.validateTournamentStatus() already throws 400 on non-SCHEDULED tournaments. No code change needed there.

**advanceBracketSlot(tx, updatedMatch, winnerId)**
- Guard 1: if (!updatedMatch.roundId) return — not a bracket match, skip
- Guard 2: if (updatedMatch.isBye) return — BYE advancement handled at bracket creation, skip
- Reads current round: tx.round.findUnique({ where: { id: updatedMatch.roundId }, select: { roundNumber, bracketId, tournamentId } })
- If no current round: return
- Finds next round: tx.round.findFirst({ where: { tournamentId: currentRound.tournamentId, bracketId: currentRound.bracketId, roundNumber: currentRound.roundNumber + 1 } })
- If no next round: return — this was the final round, no slot to populate
- Computes next match slot:
  - nextMatchNumber = Math.ceil(updatedMatch.matchNumber / 2)
  - isOdd = updatedMatch.matchNumber % 2 === 1
- Finds target: tx.match.findFirst({ where: { roundId: nextRound.id, matchNumber: nextMatchNumber } })
- If no target: return
- Updates: tx.match.update({ where: { id: nextMatch.id }, data: isOdd ? { player1Id: winnerId } : { player2Id: winnerId } })
- For doubles: also advance pair slot. Derive which pair won: if resultJson.winner === 'PLAYER1', advance pair1Id; if 'PLAYER2', advance pair2Id. The pair1Id/pair2Id from updatedMatch should be included in the select when called (Task 2 will ensure this).
  - Update: isOdd ? { player1Id: winnerId, pair1Id: winnerPairId } : { player2Id: winnerId, pair2Id: winnerPairId }
  - winnerPairId is updatedMatch.pair1Id if winner is PLAYER1, updatedMatch.pair2Id if PLAYER2. May be null for singles — that is fine (Prisma will write null and existing pair fields stay unchanged for singles).

**checkAndCompleteTournament(tx, tournamentId, isOrganizer)**
- Guard: if (!isOrganizer) return — only organizer result triggers COMPLETED
- Counts incomplete MAIN-bracket non-BYE matches:
  ```javascript
  const incompleteCount = await tx.match.count({
    where: {
      tournamentId,
      bracket: { bracketType: 'MAIN' },
      isBye: false,
      status: { not: 'COMPLETED' }
    }
  });
  ```
- If incompleteCount === 0: update tournament status to COMPLETED, lastStatusChange = new Date()

Use PrismaClient imported at top of file. Use the same throwError helper pattern from matchResultService (inline function, not imported).
