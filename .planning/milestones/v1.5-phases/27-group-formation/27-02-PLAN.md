**2. Create `backend/src/api/groupDrawController.js`:**

Two handlers:

`generateDraw(req, res)`:
- Extract `tournamentId` from `req.params.id`
- Extract `{ groupCount, seededRounds, randomSeed }` from `req.body`
- Call `generateGroupDraw(tournamentId, { groupCount, seededRounds, randomSeed })`
- On success: `res.status(201).json({ success: true, data: result })`
- Error mapping (follow bracketController pattern):
  - `TOURNAMENT_NOT_FOUND` → 404
  - `REGISTRATION_NOT_CLOSED` → 400
  - `INVALID_GROUP_COUNT` → 400
  - `INSUFFICIENT_PLAYERS` → 400
  - `UNBALANCED_GROUPS` → 400
  - `TOURNAMENT_LOCKED` → 409
  - Default → 500

`swapParticipants(req, res)`:
- Extract `tournamentId` from `req.params.id`
- Extract `{ participantAId, participantBId }` from `req.body`
- Call `swapGroupParticipants(tournamentId, participantAId, participantBId)` from groupPersistenceService
- On success: `res.status(200).json({ success: true, data: result })`
- Error mapping:
  - `TOURNAMENT_NOT_FOUND` → 404
  - `PARTICIPANT_NOT_FOUND` → 404
  - `SAME_GROUP_SWAP` → 400
  - `TOURNAMENT_LOCKED` → 409
  - Default → 500

**3. Create `backend/src/api/routes/groupDrawRoutes.js`:**

```javascript
import express from 'express';
import { generateDraw, swapParticipants } from '../groupDrawController.js';
import { validateBody } from '../../middleware/validate.js';
import { requireAuth, requireRole } from '../../middleware/auth.js';
import { generateGroupDrawSchema, swapGroupParticipantsSchema } from '../validators/groupDrawValidator.js';

const router = express.Router();

// POST /api/v1/tournaments/:id/group-draw
router.post(
  '/:id/group-draw',
  requireAuth,
  requireRole(['ORGANIZER', 'ADMIN']),
  validateBody(generateGroupDrawSchema),
  generateDraw
);

// POST /api/v1/tournaments/:id/group-draw/swap
router.post(
  '/:id/group-draw/swap',
  requireAuth,
  requireRole(['ORGANIZER', 'ADMIN']),
  validateBody(swapGroupParticipantsSchema),
  swapParticipants
);

export default router;
```
