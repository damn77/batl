import { PrismaClient } from '@prisma/client';

function makeError(code, message) {

  err.code = code;
  return err;

**closeRegistration(tournamentId):**
- Find tournament by id (select: { status, registrationClosed })
- If not found: throw makeError('TOURNAMENT_NOT_FOUND', ...)
- If registrationClosed is already true: throw makeError('ALREADY_CLOSED', ...)
- Update tournament: `{ registrationClosed: true }`
- Return updated tournament

**generateBracket(tournamentId, options = {}):**
Options: `{ randomSeed?, doublesMethod? }` where doublesMethod = 'PAIR_SCORE' | 'AVERAGE_SCORE' (default: 'PAIR_SCORE')

Steps:
1. Load tournament: `{ id, status, registrationClosed, categoryId, category: { type } }`
2. Guard: tournament not found → TOURNAMENT_NOT_FOUND
3. Guard: registrationClosed is false → REGISTRATION_NOT_CLOSED
4. Guard: status is IN_PROGRESS or COMPLETED → BRACKET_LOCKED

5. Load registered players:
   - If category.type === 'SINGLES': query TournamentRegistration with status: 'REGISTERED', include player
   - If category.type === 'DOUBLES': query PairRegistration with status: 'REGISTERED', include pair

6. Count entities (playerCount for singles, pairCount for doubles). If fewer than 4: throw makeError('INSUFFICIENT_PLAYERS', 'Need at least 4 players to generate bracket').

7. For doubles with doublesMethod === 'AVERAGE_SCORE': compute average seeding score for each pair by looking up each player's individual RankingEntry for the category/year. Pairs with missing entries score 0. Sort pairs descending by averageScore. This sorted list becomes the seeding order passed to generateSeededBracket (override the service's internal ranking lookup by passing unseededPlayers if supported, otherwise handle by building a custom player list sorted by score).

   For simplicity, for AVERAGE_SCORE doubles: construct the players array sorted by averageScore descending and pass it as the seeding order. The existing generateSeededBracket() uses categoryId to look up rankings; for AVERAGE_SCORE we need to compute scores ourselves and pass them in. Since the service doesn't support pre-computed scores, compute the sorted order and pass the pair IDs as unseededPlayers in the correct rank position — OR simply call `generateSeededBracket(categoryId, pairCount, randomSeed)` and rely on Feature 008 PAIR rankings for PAIR_SCORE mode, and for AVERAGE_SCORE mode, compute the average and sort, then build a custom positions array without calling generateSeededBracket (manual placement per the algorithm: seed 1 at top, seed 2 at bottom, etc.). Keep this implementation note for now — implement PAIR_SCORE mode first and leave AVERAGE_SCORE as a stub that falls through to PAIR_SCORE.

8. Call `generateSeededBracket(categoryId, playerCount, randomSeed)` → get `{ bracket: { positions, bracketSize } }`

9. Inside `prisma.$transaction(async (tx) => { ... })`:
   a. Delete in cascade order:
      ```javascript
      await tx.match.deleteMany({ where: { tournamentId } });
      await tx.round.deleteMany({ where: { tournamentId } });
      await tx.bracket.deleteMany({ where: { tournamentId } });
      ```
   b. Create Bracket:
      ```javascript
      const bracket = await tx.bracket.create({
        data: { tournamentId, bracketType: 'MAIN', matchGuarantee: 'MATCH_1' }
      });
      ```
   c. Create Rounds + Matches:
      - totalRounds = Math.log2(bracketSize)
      - matchNumber counter starting at 1
      - For roundNum = 1 to totalRounds:
        - Create Round: `{ tournamentId, bracketId: bracket.id, roundNumber: roundNum }`
        - If roundNum === 1: create matches from positions pairs (pos[i*2], pos[i*2+1])
          - isByeMatch = pos1.isBye || !pos2 || pos2.isBye
          - For singles: player1Id = pos1.entityId || null, player2Id = isByeMatch ? null : pos2.entityId
          - For doubles: pair1Id = pos1.entityId || null, pair2Id = isByeMatch ? null : pos2.entityId, player1Id = null, player2Id = null
        - If roundNum > 1: create bracketSize/2^roundNum placeholder matches (all ids null, status: SCHEDULED)

10. Return `{ bracket, roundCount: totalRounds, matchCount }`

**Key pitfall:** For later-round matches, BOTH player1Id and player2Id must be null. Do not set them to any sentinel value — the nullable migration from Plan 01 handles this.
