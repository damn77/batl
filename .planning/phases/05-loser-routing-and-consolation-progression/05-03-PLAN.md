**Business rules (from CONTEXT.md):**
- Either the organizer or the player (logged-in, self-service) can record a consolation opt-out
- Opt-out is allowed any time before the player's next consolation match is played (COMPLETED)
- For doubles: opt-out is pair-level — if one partner opts out, the whole pair withdraws from consolation
- Pre-placement opt-out is valid: if a player opts out before their main bracket match is resolved, the ConsolationOptOut record is created immediately; routeLoserToConsolation checks it at placement time

**Exported function: `recordOptOut`**

```javascript
/**
 * Record a consolation opt-out for a player or pair in a tournament.
 *
 * @param {Object} params
 * @param {string} params.tournamentId - UUID of the tournament
 * @param {string|null} params.playerId - PlayerProfile ID (singles) or null (doubles)

 * @returns {Promise<Object>} Created ConsolationOptOut record
 * @throws {Error} TOURNAMENT_NOT_FOUND | NOT_CONSOLATION_TOURNAMENT | ALREADY_OPTED_OUT | NEXT_MATCH_ALREADY_PLAYED | NOT_AUTHORIZED
 */
```

**Validation steps inside `recordOptOut`:**

1. Fetch tournament (select `id`, `formatConfig`) — throw 404 `TOURNAMENT_NOT_FOUND` if not found.
2. Parse `formatConfig` JSON, check `matchGuarantee === 'MATCH_2'`. If not MATCH_2, throw 400 `NOT_CONSOLATION_TOURNAMENT` ("This tournament does not have a consolation bracket").
3. Check authorization:
   - If `isOrganizer`: allowed for any player/pair
   - If not organizer (player self-service):
     - Singles: `submitterPlayerId` must equal `playerId`. Throw 403 `NOT_AUTHORIZED` if not.
     - Doubles: fetch the DoublesPair where `id=pairId`, check that `submitterPlayerId` is `player1Id` or `player2Id` of the pair. Throw 403 `NOT_AUTHORIZED` if not a pair member.
4. Check for existing opt-out: `prisma.consolationOptOut.findFirst({ where: { tournamentId, OR: [{ playerId }, { pairId }].filter(Boolean) } })`. If found, throw 409 `ALREADY_OPTED_OUT` ("Player/pair has already opted out of consolation").
5. Check if the next consolation match is already played:
   - Find the CONSOLATION bracket for this tournament.
   - If no consolation bracket yet (draw not generated), skip this check — pre-draw opt-out is valid.
   - If consolation bracket exists: find matches where the player/pair is already placed (player1Id/player2Id = playerId OR pair1Id/pair2Id = pairId) and status = 'COMPLETED'. If any exist, throw 409 `NEXT_MATCH_ALREADY_PLAYED` ("Cannot opt out — consolation match has already been played").
6. Create the opt-out record:
```javascript
const optOut = await prisma.consolationOptOut.create({
  data: {
    tournamentId,
    playerId: playerId || null,
    pairId: pairId || null,
    recordedBy: isOrganizer ? 'ORGANIZER' : 'SELF'
  }
});
return optOut;
```

**Create `backend/src/api/consolationOptOutController.js`:**

```javascript
/**
 * Consolation Opt-Out Controller
 *
 * Handles POST /api/v1/tournaments/:id/consolation-opt-out
 * Delegates business logic to consolationOptOutService.
 *
 * Phase: 05-loser-routing-and-consolation-progression
 * Requirement: LIFE-05
 */

import { recordOptOut } from '../services/consolationOptOutService.js';

/**
 * POST /api/v1/tournaments/:id/consolation-opt-out
 *
 * Request body:
 *   - playerId: string (required for singles; omit for doubles)
 *   - pairId: string (required for doubles; omit for singles)
 *
 * Authorization:
 *   - ORGANIZER/ADMIN: can opt out any player/pair
 *   - PLAYER: can only opt out themselves (singles) or their pair (doubles)
 */
export async function recordConsolationOptOut(req, res, next) {
  try {
    const tournamentId = req.params.id;
    const isOrganizer = ['ADMIN', 'ORGANIZER'].includes(req.user.role);
    const submitterPlayerId = req.user.playerId;
    const { playerId, pairId } = req.body;

    if (!playerId && !pairId) {
      return res.status(400).json({
        success: false,
        error: { code: 'MISSING_ENTITY', message: 'Provide either playerId (singles) or pairId (doubles)' }
      });
    }

    if (playerId && pairId) {
      return res.status(400).json({
        success: false,
        error: { code: 'AMBIGUOUS_ENTITY', message: 'Provide playerId OR pairId, not both' }
      });
    }

    const optOut = await recordOptOut({
      tournamentId,
      playerId: playerId || null,
      pairId: pairId || null,
      isOrganizer,
      submitterPlayerId
    });

    return res.status(201).json({ success: true, data: optOut });
  } catch (err) {
    if (err.statusCode) {
      return res.status(err.statusCode).json({
        success: false,
        error: { code: err.code, message: err.message }
      });
    }
    next(err);
  }
}
```
