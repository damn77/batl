<objective>
Extend the backend point calculation system to support GROUP and COMBINED format tournaments.

Purpose: Currently, point calculation only works for KNOCKOUT tournaments where the organizer supplies a results array. GROUP and COMBINED formats need the backend to auto-derive results from group standings positions, award points using group size (not total participants) as the participant count, and apply an offset-based tier hierarchy for COMBINED tournaments that guarantees worst main bracket result > best secondary bracket result > best group-only placement.

Output: Extended pointCalculationService with group-aware point functions, extended calculate-points route handler with auto-derive branch, unit and integration tests.
</objective>

    ```javascript
    export function computeTierOffsets(largestGroupSize, pointConfig, secondaryBracketSize) {
      const { multiplicativeValue = 2, doublePointsEnabled = false } = pointConfig;
      // Max group points = 1st place in largest group
      const maxGroupPoints = calculatePlacementPoints(largestGroupSize, 1, multiplicativeValue, doublePointsEnabled);
      const secondaryOffset = maxGroupPoints + 1;
      // Max secondary points = 1st place in secondary bracket + secondary offset
      const maxSecondaryPoints = secondaryBracketSize > 0
        ? calculatePlacementPoints(secondaryBracketSize, 1, multiplicativeValue, doublePointsEnabled) + secondaryOffset
        : secondaryOffset;
      const mainOffset = maxSecondaryPoints + 1;
      return { mainOffset, secondaryOffset };
    }
    ```

    ```javascript
    // After fetching tournament (line 283-289), add format-aware branching:
    const formatType = tournament.formatType;

    if (formatType === 'GROUP' || formatType === 'COMBINED') {
      // Auto-derive results from group standings (D-11)
      // Import new functions at top of handler
      const {
        deriveGroupResults, awardGroupPointsSingles, awardGroupPointsDoubles,
        computeTierOffsets, awardPointsSinglesTournament, awardPointsDoublesTournament
      } = await import('../../services/pointCalculationService.js');

      // Step A: Derive group results (throws UNRESOLVED_TIES if ties exist)
      let groupResults;
      try {
        groupResults = await deriveGroupResults(id);
      } catch (err) {
        if (err.code === 'UNRESOLVED_TIES') {
          return res.status(400).json({
            success: false,
            error: { code: 'UNRESOLVED_TIES', message: err.message, details: err.details }
          });
        }
        throw err;
      }

      const isDoubles = tournament.category.type !== 'SINGLES';
      const awardGroupFn = isDoubles ? awardGroupPointsDoubles : awardGroupPointsSingles;

      if (formatType === 'GROUP') {
        // GROUP-only: all participants get group placement points, no offset
        const awarded = await awardGroupFn(id, groupResults, pointConfig, 0);
        return res.json({
          success: true,
          data: { message: 'Points calculated and awarded successfully', tournamentId: id, resultsCount: awarded.length, calculationMethod: 'PLACEMENT' }
        });
      }

      // COMBINED: partition into advancing vs non-advancing
      // Fetch brackets to determine which players advanced
      const brackets = await prisma.bracket.findMany({
        where: { tournamentId: id },
        select: { id: true, bracketType: true }
      });

      // Get advancement mapping: which entities are in which bracket
      // Query matches in brackets to find participating player/pair IDs
      const bracketMatches = await prisma.match.findMany({
        where: { tournamentId: id, bracketId: { not: null } },
        select: { player1Id: true, player2Id: true, pair1Id: true, pair2Id: true, bracket: { select: { bracketType: true } } }
      });

      // Build set of advancing entity IDs per bracket type
      const advancedToMain = new Set();
      const advancedToSecondary = new Set();
      for (const m of bracketMatches) {
        const ids = isDoubles
          ? [m.pair1Id, m.pair2Id].filter(Boolean)
          : [m.player1Id, m.player2Id].filter(Boolean);
        if (m.bracket.bracketType === 'MAIN') ids.forEach(id => advancedToMain.add(id));
        if (m.bracket.bracketType === 'SECONDARY') ids.forEach(id => advancedToSecondary.add(id));
      }
      const allAdvanced = new Set([...advancedToMain, ...advancedToSecondary]);

      // Filter group results to non-advancing only
      const nonAdvancingGroupResults = groupResults.map(gr => ({
        ...gr,
        results: gr.results.filter(r => {
          const entityId = isDoubles ? r.pairId : r.playerId;
          return !allAdvanced.has(entityId);
        })
      })).filter(gr => gr.results.length > 0);

      // Compute tier offsets (D-07)
      const largestGroupSize = Math.max(...groupResults.map(gr => gr.groupSize));
      const secondaryBracket = brackets.find(b => b.bracketType === 'SECONDARY');
      const secondaryBracketSize = secondaryBracket
        ? await prisma.match.count({ where: { bracketId: secondaryBracket.id, roundNumber: 1, isBye: false } }) * 2
        : 0;
      // Note: bracket participant count approximated from round 1 non-bye match count * 2
      // More accurate: count distinct participants in bracket
      const offsets = computeTierOffsets(largestGroupSize, pointConfig, secondaryBracketSize || 0);

      let allAwarded = [];

      // Award group placement points to non-advancing players (offset = 0)
      if (nonAdvancingGroupResults.length > 0) {
        const groupAwarded = await awardGroupFn(id, nonAdvancingGroupResults, pointConfig, 0);
        allAwarded.push(...groupAwarded);
      }

      // Award knockout points to advancing players
      // Requires results body for knockout results (same as current KNOCKOUT flow)
      // The organizer must supply knockout results in the request body
      if (results && Array.isArray(results) && results.length > 0) {
        // Add offset based on bracket type for each result
        const knockoutResultsWithOffset = results.map(r => {
          // Determine which bracket this result belongs to
          const entityId = isDoubles ? r.pairId : r.playerId;
          const offset = advancedToMain.has(entityId) ? offsets.mainOffset
            : advancedToSecondary.has(entityId) ? offsets.secondaryOffset
            : 0;
          return { ...r, _offset: offset };
        });

        // Award knockout points with offsets
        // Use existing award functions but need to handle offset
        // Since existing awardPointsSingles/Doubles uses tournament.registrations.length as participantCount,
        // we need a different approach for COMBINED knockout:
        // Call awardGroupPointsSingles/Doubles with bracket-shaped results
        // Actually: build bracket-level groupResults for the award function
        for (const bracket of brackets) {
          const bracketEntityIds = bracket.bracketType === 'MAIN' ? advancedToMain : advancedToSecondary;
          const bracketResults = results.filter(r => {
            const entityId = isDoubles ? r.pairId : r.playerId;
            return bracketEntityIds.has(entityId);
          });
          if (bracketResults.length === 0) continue;

          const bracketParticipantCount = bracketEntityIds.size;
          const offset = bracket.bracketType === 'MAIN' ? offsets.mainOffset : offsets.secondaryOffset;

          // Use the group award function with a synthetic groupResult entry
          const syntheticGroupResult = [{
            groupId: bracket.id, // reuse bracket ID as group key
            groupSize: bracketParticipantCount,
            results: bracketResults
          }];
          const koAwarded = await awardGroupFn(id, syntheticGroupResult, pointConfig, offset);
          allAwarded.push(...koAwarded);
        }
      }

      return res.json({
        success: true,
        data: { message: 'Points calculated and awarded successfully', tournamentId: id, resultsCount: allAwarded.length, calculationMethod: 'PLACEMENT' }
      });
    }
    ```
