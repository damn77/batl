```javascript
import apiClient from './apiClient';

/**
 * Get advancement preview ŌĆö waterfall player assignments
 * @param {string} tournamentId
 * @returns {Promise<Object>} { mainSlots, secondarySlots, eliminated, mainBracketInfo, secondaryBracketInfo }
 */
export async function getAdvancementPreview(tournamentId) {
  const response = await apiClient.get(`/api/v1/tournaments/${tournamentId}/advancement/preview`);
  return response.data;
}

/**
 * Confirm advancement ŌĆö generate knockout brackets from group results
 * @param {string} tournamentId
 * @returns {Promise<Object>} { success, mainBracket, secondaryBracket }
 */
export async function confirmAdvancement(tournamentId) {
  const response = await apiClient.post(`/api/v1/tournaments/${tournamentId}/advancement`);
  return response.data;
}

/**
 * Revert advancement ŌĆö delete brackets, unlock group results
 * @param {string} tournamentId
 * @returns {Promise<Object>} { success }
 */
export async function revertAdvancement(tournamentId) {
  const response = await apiClient.delete(`/api/v1/tournaments/${tournamentId}/advancement`);
  return response.data;
}
```
