**Waterfall algorithm (`computeWaterfall` internal function):**
```javascript
function computeWaterfall(allGroupStandings, mainN, secondaryM) {
  // allGroupStandings: array of { groupId, groupNumber, standings: [{entity, position, wins, setDiff, gameDiff, totalGames, ...}] }
  // Flatten all standings into position buckets
  const byPosition = new Map(); // position → [{entity, groupNumber, wins, setDiff, gameDiff, totalGames}]
  for (const gs of allGroupStandings) {
    for (const entry of gs.standings) {
      if (!byPosition.has(entry.position)) byPosition.set(entry.position, []);
      byPosition.get(entry.position).push({
        entity: entry.entity,
        groupId: gs.groupId,
        groupNumber: gs.groupNumber,
        position: entry.position,
        wins: entry.wins,
        setDiff: entry.setDiff,
        gameDiff: entry.gameDiff,
        totalGames: entry.totalGames,
      });
    }
  }

  const mainSlots = [];
  const secondarySlots = [];
  let mainRemaining = mainN;
  let secondaryRemaining = secondaryM;
  const eliminated = [];

  // Iterate positions in ascending order (1st place first)
  const sortedPositions = [...byPosition.keys()].sort((a, b) => a - b);

  for (const position of sortedPositions) {
    const playersAtPosition = byPosition.get(position);

    // Fill main bracket first
    if (mainRemaining > 0) {
      if (playersAtPosition.length <= mainRemaining) {
        mainSlots.push(...playersAtPosition.map(p => ({ ...p, isSpillover: false, bracket: 'main' })));
        mainRemaining -= playersAtPosition.length;
      } else {
        const ranked = crossGroupRank(playersAtPosition);
        const toMain = ranked.slice(0, mainRemaining);
        mainSlots.push(...toMain.map((p, i) => ({ ...p, isSpillover: toMain.length < playersAtPosition.length, bracket: 'main' })));
        const remainder = ranked.slice(mainRemaining);
        mainRemaining = 0;

        // Overflow to secondary
        if (secondaryRemaining > 0) {
          const toSecondary = remainder.slice(0, secondaryRemaining);
          secondarySlots.push(...toSecondary.map(p => ({ ...p, isSpillover: true, bracket: 'secondary' })));
          secondaryRemaining -= toSecondary.length;
          eliminated.push(...remainder.slice(toSecondary.length).map(p => ({ ...p, bracket: null })));
        } else {
          eliminated.push(...remainder.map(p => ({ ...p, bracket: null })));
        }
      }
    } else if (secondaryRemaining > 0) {
      if (playersAtPosition.length <= secondaryRemaining) {
        secondarySlots.push(...playersAtPosition.map(p => ({ ...p, isSpillover: false, bracket: 'secondary' })));
        secondaryRemaining -= playersAtPosition.length;
      } else {
        const ranked = crossGroupRank(playersAtPosition);
        secondarySlots.push(...ranked.slice(0, secondaryRemaining).map(p => ({ ...p, isSpillover: true, bracket: 'secondary' })));
        eliminated.push(...ranked.slice(secondaryRemaining).map(p => ({ ...p, bracket: null })));
        secondaryRemaining = 0;
      }
    } else {
      eliminated.push(...playersAtPosition.map(p => ({ ...p, bracket: null })));
    }
  }

  return { mainSlots, secondarySlots, eliminated };
}
```

**Cross-group ranking function (5-level, NO H2H):**
```javascript
function crossGroupRank(players) {
  return [...players].sort((a, b) => {
    if (b.wins !== a.wins) return b.wins - a.wins;
    if (b.setDiff !== a.setDiff) return b.setDiff - a.setDiff;
    if (b.gameDiff !== a.gameDiff) return b.gameDiff - a.gameDiff;
    if (a.totalGames !== b.totalGames) return a.totalGames - b.totalGames; // fewer = better
    return a.entity.name.localeCompare(b.entity.name);
  });
}
```

**Seeding from group standings:**
```javascript
// Build seed list from waterfall slots
function buildSeedList(slots) {
  // Slots are already ordered: 1st-place finishers first (by cross-group rank), then 2nd-place, etc.
  // This IS the seed order — position 1 in slots = seed 1
  return slots.map((slot, index) => ({
    seed: index + 1,
    entityId: slot.entity.id,
    name: slot.entity.name
  }));
}

// Generate positions array using seeding placement functions
function generateSeededPositions(seedList, bracketTemplate) {
  const { bracketSize, structure } = bracketTemplate;
  const seedCount = getSeedCount(seedList.length);
  const seeds = seedList.slice(0, seedCount);
  const randomSeed = createRandomSeed();

  // Select placement function based on seed count
  let positions;
  if (seedCount <= 2) positions = placeTwoSeeds(bracketSize, seeds, structure);
  else if (seedCount <= 4) positions = placeFourSeeds(bracketSize, seeds, randomSeed, structure);
  else if (seedCount <= 8) positions = placeEightSeeds(bracketSize, seeds, randomSeed, structure);
  else positions = placeSixteenSeeds(bracketSize, seeds, randomSeed, structure);

  // Place remaining (unseeded) players randomly in empty positions
  const unseeded = seedList.slice(seedCount);
  const emptyPositions = positions
    .map((p, i) => ({ ...p, index: i }))
    .filter(p => !p.entityId && !p.isBye);
  const shuffledEmpty = shuffle(emptyPositions.map(p => p.index), randomSeed + '-unseeded');

  for (let i = 0; i < unseeded.length && i < shuffledEmpty.length; i++) {
    positions[shuffledEmpty[i]] = {
      ...positions[shuffledEmpty[i]],
      entityId: unseeded[i].entityId,
      seed: null
    };
  }

  return { positions, randomSeed };
}
```
